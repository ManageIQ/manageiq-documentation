= Set the Owner of a Virtual Machine

Request: 

----
POST /api/vms/1921
----

[source]
----
{
  "action": "set_owner",
  "resource" : {
    "owner" : "admin"
  }
}
----

Response: 

[source]
----
{
  "success": true,
  "message": "VM id:1921 name:'aab_demo_vm' setting owner to 'admin'",
  "href": "https://hostname/api/vms/1921"
}
----
